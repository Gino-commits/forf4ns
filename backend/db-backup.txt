create database forf4ns;
use forf4ns;
CREATE TABLE Usertypes(id INT NOT NULL AUTO_INCREMENT, type VARCHAR(30) UNIQUE NOT NULL, CONSTRAINT PK_USERTYPE PRIMARY KEY(id));
CREATE TABLE Users(id BINARY(16) DEFAULT(UUID_TO_BIN(UUID())), username VARCHAR(20) UNIQUE NOT NULL, password VARCHAR(255) NOT NULL, usertype INT NOT NULL, CONSTRAINT PK_USER PRIMARY KEY(id), CONSTRAINT FK_USERTYPE FOREIGN KEY(usertype) REFERENCES Usertypes(id));
